题目：输入一个链表的头结点，从尾到头反过来打印出每个节点的值。

```
import java.util.Stack;

public class AddToTail {

/**
 * @ClassName: AddToTail
 * @Description:
 * @Version 1.0
 * Created by Fanchao.Zhao on 2019/07/07-9:24.
 */

static class Node{
       int data;
       Node next;

       public Node(int data, Node next) {
           this.data = data;
           this.next = next;
       }
   }
   public static void main(String[] args){

       Node node1 = new Node(5,null);
       Node node2 = new Node(4,node1);
       Node head = node2;
       while (head.next!=null){
           head = head.next;
       }
       for (int i=0;i<5;i++){
           Node node3 = new Node(3+i,null);
           head.next = node3;
           head = head.next;
       }
       Node a = node2;
       while (a.next!=null){
           System.out.print(" "+a.data+" ");
           a = a.next;
       }
       System.out.print(" "+a.data+" ");
       System.out.println();
       printInResversedOrder(node2);
       System.out.println();
       printInResversedOrderByStack(node2);
   }
   public static void printInResversedOrder(Node node){
        Node head = node;
        if (head != null){
            if (head.next != null){
                printInResversedOrder(head.next);
            }
        }
       System.out.print(" "+head.data+" ");
   }
    public static void printInResversedOrderByStack(Node node){
        Node head = node;
        Stack stack = new Stack();
        while (head.next!=null){
            stack.push(head.data);
            head = head.next;
        }
        stack.push(head.data);
        while (!stack.isEmpty()){
            System.out.print(" "+stack.pop()+" ");
        }
    }

}
```
