import java.util.HashSet;
import java.util.Set;

public class Q3_Duplicate {

/**
 * @Description:  查询数组中重复的数字
 * Created by shanzhihong on 2019/07/02-10:31.
 * @version 1.0
 */
    public Set<Integer> getDuplicate(int[] arr){
        Set<Integer> set = new HashSet<>();
        int[] arr1 = new int[arr.length];
        for (int i=0;i<arr1.length;i++){
            arr1[i] = 0;
        }
        if (arr.length < 0){
            return set;
        }else {
            for (int i=0;i<arr.length;i++){
                if (arr1[arr[i]] == 0){
                    arr1[arr[i]] = 1;
                }else {
                    set.add(arr[i]);
                }
            }
            return set;
        }
    }
    public void swap(int[] arr, int a, int b){
        int temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }
    public static void main(String[] args){
        int[] arr = new int[]{1,2,3,4,2,1,4,2,5};
        Q3_Duplicate a = new Q3_Duplicate();
        Set<Integer> set = a.getDuplicate(arr);
        for (int i:set){
            System.out.println(i);
        }
    }
}
